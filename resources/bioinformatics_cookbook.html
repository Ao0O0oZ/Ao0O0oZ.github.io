scBCR Dandelion 预处理

从 10x VDJ-B contig 到 Dandelion 对象、克隆和多样性。样本命名和个体分组按项目实际情况替换。

BCRDandelion
import os
import shutil
import pandas as pd
import scanpy as sc
import dandelion as ddl
import muon as mu

PROJECT = "/path/to/project"
NPROC = "32"
os.environ["GERMLINE"] = "/path/to/dandelion/germlines"
os.environ["IGDATA"] = "/path/to/dandelion/igblast"
os.environ["BLASTDB"] = "/path/to/dandelion/blast"

def collect_bcr_contigs(cellranger_out, dandelion_dir):
    samples = []
    for sample in os.listdir(cellranger_out):
        src = f"{cellranger_out}/{sample}/outs/multi/vdj_b"
        dst = f"{dandelion_dir}/{sample}"
        os.makedirs(dst, exist_ok=True)
        try:
            shutil.copy(f"{src}/all_contig.fasta", dst)
            shutil.copy(f"{src}/all_contig_annotations.csv", dst)
            samples.append(sample)
        except FileNotFoundError:
            print(f"No VDJ-B output: {sample}")
    return samples

sample_list = collect_bcr_contigs(
    f"{PROJECT}/cellranger/out",
    f"{PROJECT}/bcr/dandelion",
)

ddl.pp.format_fastas(sample_list, prefix=sample_list, filename_prefix="all")
ddl.pp.reannotate_genes(
    sample_list, org="human", db="imgt", filename_prefix="all",
    additional_args={"igblastn": ["-num_threads", NPROC]},
)
ddl.pp.assign_isotypes(sample_list, filename_prefix="all", save_plot=True, show_plot=False)

# Optional: one individual can contain multiple time points/samples
sample_group = {"individual_1": ["sample_A", "sample_B"]}
for individual, samples in sample_group.items():
    ddl.pp.reassign_alleles(
        data=samples,
        combined_folder=f"{individual}_combine",
        filename_prefix="all",
        org="human",
        db="imgt",
    )

adata = sc.read(f"{PROJECT}/data/rna_merge_qc.h5ad")
bcr = pd.concat(
    pd.read_csv(f"{PROJECT}/bcr/dandelion/{s}/dandelion/all_contig_dandelion.tsv", sep="\t")
    for s in sample_list
)

cells = sorted(set(adata.obs_names) & set(bcr["cell_id"]))
adata = adata[cells].copy()
bcr = bcr.loc[bcr["cell_id"].isin(cells)]

vdj, adata = ddl.pp.check_contigs(
    data=bcr, adata=adata, library_type="ig",
    productive_only=True, filter_extra=False,
)
ddl.tl.find_clones(vdj)
ddl.pp.quantify_mutations(vdj, combine=False, split_locus=True, nproc=32)
for method in ["gini", "chao1", "shannon"]:
    ddl.tl.clone_diversity(vdj, groupby="sample_id", method=method)

ddl.tl.transfer(adata, vdj)
mdata = ddl.to_scirpy(vdj, to_mudata=True, gex_adata=adata)
mu.pp.intersect_obs(mdata)
mdata.write(f"{PROJECT}/data/rna_bcr_filter.h5mu")
vdj.write_h5ddl(f"{PROJECT}/data/bcr_dandelion.h5ddl")

def add_rna_qc(adata):
    adata.var["mt"] = adata.var_names.str.startswith("MT-")
    adata.var["ribo"] = adata.var_names.str.startswith(("RPS", "RPL"))
    adata.var["mtribo"] = adata.var_names.str.startswith(("MRPS", "MRPL"))

        adata,
        qc_vars=["mt", "ribo", "mtribo", "heatshock", "hb"],
        inplace=True,
        percent_top=[20],
        log1p=True,
    )
    return adata

adata = add_rna_qc(adata)
adata = adata[
    (adata.obs["total_counts"] > 500)
    & (adata.obs["total_counts"] < 20000)

scRNA Harmony 整合

先保留 counts layer，再用 Seurat v3 HVG、PCA、Harmony、neighbors、UMAP。

RNAHarmony
import scanpy as sc
import pandas as pd

rna = sc.read("data/multiome_rna_merge_qc.h5ad")
sc.pp.normalize_total(rna, target_sum=1e4)
sc.pp.log1p(rna)

def find_hvg(adata, n_top_genes=3000, batch_key="batch", span=0.5):
    exclude = ["mt", "ribo", "mtribo", "heatshock"]
    genes = adata.var_names[~adata.var[exclude].any(axis=1)]
    hvg = sc.pp.highly_variable_genes(
        adata[:, genes],
        n_top_genes=n_top_genes,
        flavor="seurat_v3",
        batch_key=batch_key,
        inplace=False,
        layer="counts",
        span=span,
    )
    res = pd.Series(False, index=adata.var_names)
    res.loc[hvg.index[hvg.highly_variable]] = True
    return res

rna.var["highly_variable"] = find_hvg(rna)
sc.pp.scale(rna, max_value=10)
sc.tl.pca(rna, use_highly_variable=True, random_state=123)
sc.external.pp.harmony_integrate(
    rna,
    key=["dataset", "batch"],
    basis="X_pca",
    theta=[1, 1],
    lamb=[1, 1],
    max_iter_harmony=15,
    random_state=123,
)
sc.pp.neighbors(rna, n_neighbors=20, n_pcs=30, use_rep="X_pca_harmony", random_state=123)
sc.tl.umap(rna, min_dist=0.2, random_state=123)

CellTypist 自动注释

先从 counts 标准化到 1e4，再 log1p。CellTypist 对 dense matrix 兼容性更好。

CellTypistannotation
import scanpy as sc
import celltypist
from celltypist import models

def celltypist_predict(adata, model_name="Immune_All_High.pkl"):
    x = adata.copy()
    x.X = x.layers["counts"].copy()
    sc.pp.normalize_total(x, target_sum=1e4)
    sc.pp.log1p(x)
    if hasattr(x.X, "toarray"):
        x.X = x.X.toarray()

    model = models.Model.load(model=model_name)
    pred = celltypist.annotate(x, model=model, majority_voting=True)
    return pred.to_adata().copy()

pred_adata = celltypist_predict(adata)
adata.obs["celltypist_celltype"] = pred_adata.obs["majority_voting"].reindex(adata.obs_names)

Jaccard Index

用于比较两个细胞集合、cluster overlap 或拆样 assignment 的一致性。

metric
def jaccard(a, b):
    a, b = set(a), set(b)
    union = a | b
    if len(union) == 0:
        return 0
    return len(a & b) / len(union)

空间分析

图像交互对齐

用于 HE、ssDNA、IF 图像粗配准。函数返回叠加图；在 notebook 中配合 ipywidgets.interactive 调参。

imagealignment
import numpy as np
import matplotlib.pyplot as plt
from PIL import Image, ImageEnhance
from ipywidgets import interactive

def tint_image(img, color, brightness=2):
    img = np.asarray(img.convert("RGB")).copy()
    if color == "cyan":
        img[:, :, 0] = 0.1 * img[:, :, 0]
    elif color == "cherry":
        img[:, :, 1] = 0.094 * img[:, :, 1]
        img[:, :, 2] = 0.266 * img[:, :, 2]
    return ImageEnhance.Brightness(Image.fromarray(img)).enhance(brightness)

def place_image(img, canvas_size, scale=1, angle=0, x_offset=0, y_offset=0):
    canvas = Image.fromarray(np.zeros((canvas_size, canvas_size, 3), dtype=np.uint8))
    w, h = img.size
    img = img.resize((round(w * scale), round(h * scale))).rotate(angle, expand=True)
    pos = ((canvas.width - img.width) // 2, (canvas.height - img.height) // 2)
    canvas.paste(img, pos)
    return canvas.transform(canvas.size, Image.AFFINE, (1, 0, x_offset, 0, 1, y_offset))

def image_align(img_1, img_2, x_offset_1=0, y_offset_1=0, angle_1=0,
                x_offset_2=0, y_offset_2=0, angle_2=0, scale_1=1, scale_2=1):
    img_1 = tint_image(img_1, "cherry")
    img_2 = tint_image(img_2, "cyan")
    canvas_size = int(max(*img_1.size, *img_2.size))
    img_1 = place_image(img_1, canvas_size, scale_1, angle_1, x_offset_1, y_offset_1)
    img_2 = place_image(img_2, canvas_size, scale_2, angle_2, x_offset_2, y_offset_2)

    fig, ax = plt.subplots(figsize=(10, 10))
    ax.imshow(img_1)
    ax.imshow(img_2, alpha=0.5)
    ax.set(xlim=[0, canvas_size], ylim=[canvas_size, 0], xticks=[], yticks=[])
    return fig, ax

Otsu 分割计算组织面积

适合 HE/ssDNA 图像快速估计组织区域面积。

Otsuimage
import cv2
import numpy as np

def calculate_tissue_area(img_bgr):
    gray = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY)
    blurred = cv2.GaussianBlur(gray, (5, 5), 0)
    _, mask = cv2.threshold(blurred, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    mask = cv2.bitwise_not(mask)
    area = int(np.sum(mask == 255))
    return mask, area

Ripley H 函数计算空间共定位

输入可以是同类或双变量细胞间距离矩阵。A 是组织面积或可分析区域面积。

Ripleycolocalization
import numpy as np

def ripley_h(dist_mat, r, area, bivariate=True):
    if bivariate:
        indicator_sum = np.sum(dist_mat < r)
    else:
        indicator_sum = np.sum(dist_mat < r) - dist_mat.shape[0]
    n_x, n_y = dist_mat.shape
    k = indicator_sum * area / (n_x * n_y)
    return np.sqrt(k / np.pi) - r

def ripley_h_auc(dist_mat, r_max, area, bivariate=True, step=1):
    radii = np.arange(1, r_max, step)
    values = np.array([ripley_h(dist_mat, r, area, bivariate) for r in radii])
    auc = values.sum() * step
    return radii, values, auc

置换检验计算细胞类型空间共定位

按 sample 独立计算邻域组成，再随机打乱 cell type 生成 null distribution。

KDTreepermutation
import numpy as np
import pandas as pd
from sklearn.neighbors import KDTree
from joblib import Parallel, delayed

def compute_neighbors(obs, radius=100):
    tree = KDTree(obs[["x", "y"]].values)
    return tree.query_radius(obs[["x", "y"]].values, r=radius)

def neighbor_composition(obs, groupby, neighbors):
    obs[groupby] = obs[groupby].astype("category")
    labels = obs[groupby].cat.codes.values
    names = obs[groupby].cat.categories
    k = len(names)
    sum_prop = np.zeros((k, k), dtype=np.float32)
    count = np.zeros(k, dtype=np.int32)

    for i, nbrs in enumerate(neighbors):
        src = labels[i]
        nbrs = nbrs[nbrs != i]
        if len(nbrs) == 0:
            continue
        sum_prop[src] += np.bincount(labels[nbrs], minlength=k) / len(nbrs)
        count[src] += 1

    avg = np.nan_to_num(sum_prop / count[:, None])
    res = pd.DataFrame(avg, index=names, columns=names).stack().reset_index()
    res.columns = ["source", "target", "prop"]
    return res

def permutation_once(obs, groupby, neighbors):
    perm = obs[[groupby]].copy()
    perm[groupby] = pd.Categorical(
        np.random.permutation(perm[groupby].values),
        categories=obs[groupby].cat.categories,
    )
    return neighbor_composition(perm, groupby, neighbors)["prop"].values

def colocalization_permutation(adata, sample_key="sample", groupby="cluster", radius=100, n_perm=1000, n_jobs=8):
    out = []
    for sample in adata.obs[sample_key].unique():
        obs = adata.obs.loc[adata.obs[sample_key] == sample, ["x", "y", groupby]].copy()
        obs[groupby] = obs[groupby].astype("category").cat.remove_unused_categories()
        neighbors = compute_neighbors(obs, radius)
        observed = neighbor_composition(obs, groupby, neighbors)
        perms = Parallel(n_jobs=n_jobs)(
            delayed(permutation_once)(obs, groupby, neighbors) for _ in range(n_perm)
        )
        perms = np.vstack(perms)
        observed["mean_perm"] = perms.mean(axis=0)
        observed["std_perm"] = perms.std(axis=0)
        observed["z_score"] = (observed["prop"] - observed["mean_perm"]) / observed["std_perm"]
        observed["p_empirical"] = (np.sum(perms >= observed["prop"].values, axis=0) + 1) / (n_perm + 1)
        observed[sample_key] = sample
        out.append(observed)
    return pd.concat(out, axis=0)

def stack_barplot(data, x, y, hue, palette, ax, width=0.7, reverse=True):
    df = data.copy()
    df[x] = pd.Categorical(df[x], ordered=True)

    mat = df.pivot(index=x, columns=hue, values=y).reindex(index=x_order, columns=hue_order).fillna(0)

    for label in hue_order:
        vals = mat[label].values
        ax.bar(x=x_order, height=vals, bottom=bottom, width=width, color=palette[label], label=label)
        bottom += vals

    ax.legend(loc="upper left", frameon=False, bbox_to_anchor=(0.95, 1), ncols=2, fontsize=8)
    ax.grid(False)
    return ax

import numpy as np
import seaborn as sns
from scipy import stats
from itertools import combinations

def p_to_star(p):
    if np.isnan(p) or p > 0.05:
        return "ns"
    if p > 0.01:
        return "*"
    if p > 0.001:
        return "**"
    if p > 0.0001:
        return "***"
    return "****"

def stat_test(data1, data2, method="t-test", alternative="two-sided"):
    if method == "t-test":
        _, p = stats.ttest_ind(data1, data2, equal_var=False, alternative=alternative, nan_policy="omit")
    elif method in ["wilcoxon", "mannwhitney"]:
        _, p = stats.mannwhitneyu(data1, data2, alternative=alternative)
    else:
        raise ValueError("method should be 't-test' or 'mannwhitney'")
    return p, p_to_star(p)

def boxplot_with_p(df, x, y, ax, hue=None, pairs=None, order=None, hue_order=None,
                   palette=None, method="t-test", x_rotation=90):
    sns.boxplot(data=df, x=x, y=y, hue=hue, order=order, hue_order=hue_order,
                palette=palette, ax=ax, width=0.8, linewidth=0.8, flierprops={"marker": "."})
    ax.grid(False)
    ax.tick_params(axis="x", rotation=x_rotation)

    if pairs is None:
        values = df[hue].dropna().unique() if hue else df[x].dropna().unique()
        pairs = list(combinations(values, 2))

    y_min, y_max = df[y].min(), df[y].max()
    step = 0.08 * (y_max - y_min)
    y0 = y_max + step

    if hue is None:
        labels = [t.get_text() for t in ax.get_xticklabels()]
        xpos = {lab: i for i, lab in enumerate(labels)}
        for i, (a, b) in enumerate(pairs):
            p, mark = stat_test(df.loc[df[x] == a, y], df.loc[df[x] == b, y], method=method)
            yy = y0 + i * step
            ax.plot([xpos[a], xpos[a], xpos[b], xpos[b]], [yy, yy + step / 4, yy + step / 4, yy], c="k", lw=0.8)
            ax.text((xpos[a] + xpos[b]) / 2, yy + step / 4, mark, ha="center", va="bottom")
    return ax

复合 Heatmap

主 heatmap 加顶部 column sum 和右侧 row sum，适合 CellChat pathway-LR count 展示。

heatmapCellChat
import matplotlib.pyplot as plt
from matplotlib.gridspec import GridSpec

def count_joint_heatmap(data, source_col, target_col, source_label, value_col="pathway_name",
                        vmax=None, color="#c2410c", figsize=(5, 5), top=20):
    df = data.loc[data[source_col] == source_label]
    mat = (
        df[[target_col, value_col]]
        .value_counts()
        .to_frame("count")
        .reset_index()
        .pivot(index=value_col, columns=target_col, values="count")
        .fillna(0)
    )
    col_sum = mat.sum(axis=0).sort_values(ascending=False)
    row_sum = mat.sum(axis=1).sort_values(ascending=False)
    mat = mat.loc[row_sum.index, col_sum.index]
    if top:
        mat = mat.iloc[:top]
        row_sum = row_sum.iloc[:top]

    fig = plt.figure(figsize=figsize)
    gs = GridSpec(2, 2, width_ratios=[mat.shape[1], 1.5], height_ratios=[1.5, mat.shape[0]], wspace=0.02, hspace=0.02)

    ax_heat = plt.subplot(gs[1, 0])
    mesh = ax_heat.pcolormesh(mat.columns, mat.index, mat.values, cmap="Reds", shading="nearest",
                              vmin=0, vmax=vmax, edgecolors="w", lw=.5)
    ax_heat.tick_params(axis="x", labelrotation=90, left=False, bottom=False)
    ax_heat.grid(False)

    ax_col = plt.subplot(gs[0, 0], sharex=ax_heat)
    col_sum.plot(kind="bar", color=color, ax=ax_col, width=0.8)
    ax_col.grid(False)
    ax_col.spines[["top", "right"]].set_visible(False)

    ax_row = plt.subplot(gs[1, 1], sharey=ax_heat)
    row_sum.plot(kind="barh", color=color, ax=ax_row, width=0.8)
    ax_row.invert_yaxis()
    ax_row.grid(False)
    ax_row.spines[["top", "right"]].set_visible(False)

    cax = fig.add_axes([0.05, 0.05, 0.15, 0.015])
    plt.colorbar(mesh, cax=cax, orientation="horizontal", label="Counts")
    return fig, mat
